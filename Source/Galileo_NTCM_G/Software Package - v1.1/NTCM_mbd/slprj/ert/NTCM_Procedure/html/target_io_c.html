52void targetWriteDataWithErrorCheck(const MemUnit_T * data, uint32_T size) {



53   if (xilWriteData(data, size) != XIL_DATA_STREAM_SUCCESS) {



54      for (;;) {



55         /* fatal comms error */



56      }



57   }



58}

70/**



71 * Function to send "frame" toward host for all print functions : printf, vprintf...



72 **/



73static void targetWritePrintf(unsigned short targetFid,



74                              MemUnit_T responseId,



75                              MemUnit_T errorId,



76                              char* pCharBuff,



77                              uint32_T size) {



78



79



80    /* really strange behavior if here, but we have to put this...*/



81    uint16_T size16;



82    size = (uint16_T)((size > 0xFFFF) ? 0xFFFF : size);



83    size16 = (uint16_T)size;



84



85    /* send response id */



86    targetWriteDataWithErrorCheck(&responseId, sizeof(responseId));



87    /* send error id */



88    targetWriteDataWithErrorCheck(&errorId, sizeof(errorId));



89



90    /* else we don't send the file id */



91    if (targetFid != TARGET_STDOUT) {



92        const uint16_T targetFid16 = (uint16_T)targetFid; /* We only use the



93                                                           * lowermost 16 bits of



94                                                           * targetFid. unsigned



95                                                           * int is guaranteed to



96                                                           * be at least 16 bits



97                                                           * wide. See header file



98                                                           * for more information



99                                                           * on why this cast is



100                                                           * required. */



101



102        /* send file id*/



103        targetWriteDataWithErrorCheck((const MemUnit_T*)&targetFid16, sizeof(targetFid16));



104    }



105



106    /* send fprintSize */



107    targetWriteDataWithErrorCheck((MemUnit_T*)&size16, sizeof(size16));



108    if (size > 0) {



109/* send data */



110#ifdef HOST_WORD_ADDRESSABLE_TESTING



111      targetWriteDataWithCharToMemUnitWidening(&pCharBuff[0], &hostWordStdioBuffer[0], size);



112#else



113      targetWriteDataWithErrorCheck((MemUnit_T *)&pCharBuff[0], size);



114#endif



115    }



116}

119/* Forward a printf back to the host machine



120 *



121 * the sending will be cut in parts where each are bellow TARGET_IO_BUFFER_SIZE



122 * It's an improved version of targetPrintf



123 * some things are not handled from printf :



124 * - the * in the width



125 * - %n



126 * - j z t in length moditifier



127 * - %lc for character and %ls for wchar_t for string (cf



128 * http://www.cplusplus.com/reference/cstdio/printf/)



129 * - %llu %lld (will be consider as %lu %ld)



130 */



131const char target_typefiled[] = {'d', 's', 'f', 'u', 'x', '%', 'i', 'F',



132                                 'e', 'E', 'g', 'G', 'X', 'o', 'c', 'p'};



133



134/* 60 should be enough for a %xxx  */



135#define TARGET_PRINTF_TMP_BUFF_SIZE 60u



136static char targetStdioTmpBuff[TARGET_PRINTF_TMP_BUFF_SIZE];



137static char subFmt[15];



138



139static void targetPrintfKernel(MemUnit_T responseId,



140                               unsigned short targetFid,



141                               const char* fmt,



142                               va_list argptr) {



143    MemUnit_T errorId = TARGET_IO_SUCCESS;



144    char* targetFramePtr;



145    int targetStdioTmpBuffIdx;



146



147    int idx = 0;



148



149    uint16_T expectedSize = (uint16_T)0;



150    uint16_T printSize = (uint16_T)0;



151



152    uint8_T bFinished = 0;



153    uint8_T bSendNow = 0;



154    targetStdioTmpBuffIdx = 0;



155    targetFramePtr = (char*)0;



156



157    /* 01 : measure the size of the output to be sent */



158    do {



159        if (!bSendNow) {



160            if (fmt[idx] == '%') {



161



162                int subFmtIdx = 1;



163                uint8_T found = 0;



164                subFmt[0] = fmt[idx];



165                do {



166                    size_t cpt = 0;



167



168                    idx++;



169                    subFmt[subFmtIdx] = fmt[idx];



170



171                    found = 0;



172                    for (cpt = 0; cpt < sizeof(target_typefiled); cpt++) {



173                        if (subFmt[subFmtIdx] == target_typefiled[cpt]) {



174                            found = 1;



175                            subFmt[subFmtIdx + 1] = '\0';



176                            break;



177                        }



178                    }



179                    if (found) {



180                        int numCarCurr = 0;



181                        int lengthModifier = 0;



182                        /* get the length modifier if any */



183                        if (subFmtIdx > 1) {



184



185                            switch (subFmt[subFmtIdx - 1]) {



186                            case 'l':



187                                if ((subFmtIdx > 2) && (subFmt[subFmtIdx - 2] == 'l')) {



188                                    /* long long */



189                                    /* 



190                                      the long long is really special and is not well handled by every compiler the same way



191                                      all windows version doesn't no it, where's some compiler doesn't know the int64_T which 



192                                      would be required to make things easier. we will not handle it so far => lengthModifier = 1



193                                    */



194                                    lengthModifier = 1;



195                                } else {



196                                    /* long */



197                                    lengthModifier = 1;



198                                }



199                                break;



200                            case 'L':



201                                lengthModifier = 1;



202                                break;



203                            case 'h': /* integral promotion : dont care*/



204                            default:



205                                lengthModifier = 0;



206                                break;



207                            }



208                        }



209



210                        if (subFmt[subFmtIdx] != 's') {



211                            /**



212                             * Here we have to do this (sprintf) instead of calling vsprintf.



213                             * Indeed, depending on the implementation of va_list, the va_arg was



214                             * modified or not by the vsprintf function. va_copy is C99 and not



215                             * usable. Anyway as it's not good to do any assumptions on the



216                             * va_list.. implementation, it's better not using it. Sadly, it creates



217                             * drawback. for instance, now,the %*.2d would not be working



218                             */



219                            switch (subFmt[subFmtIdx]) {



220                            case 'd':



221                            case 'i': {



222                                if (lengthModifier == 0) {



223                                    int val = va_arg(argptr, int);



224                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



225                                } else {



226                                    long int val = va_arg(argptr, long int);



227                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



228                                }



229                            } break;



230



231                            case 'u':



232                            case 'x':



233                            case 'X':



234                            case 'o': {



235                                if (lengthModifier == 0) {



236                                    unsigned int val = va_arg(argptr, unsigned int);



237                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



238                                } else {



239                                    unsigned long int val = va_arg(argptr, unsigned long int);



240                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



241                                }



242                            } break;



243



244                            case 'f':



245                            case 'F':



246                            case 'E':



247                            case 'e':



248                            case 'g':



249                            case 'G': {



250                                if (lengthModifier == 0) {



251                                    double val = va_arg(argptr, double);



252                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



253                                } else { /* if(lengthModifier == 1) */



254                                    long double val = va_arg(argptr, long double);



255                                    numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



256                                }



257                                /* always promoted to double first        */



258                            } break;



259                            case 'c': {



260                                /*promotion to int for variadic, cf



261                                 * https://en.cppreference.com/w/cpp/language/variadic_arguments*/



262                                int val = va_arg(argptr, int);



263                                numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, val);



264                            } break;



265                            case 'p': {



266                                void* ptr = va_arg(argptr, void*);



267                                numCarCurr = sprintf(&targetStdioTmpBuff[0], subFmt, ptr);



268                            } break;



269                            case '%': {



270                                char val = '%';



271                                targetStdioTmpBuff[0] = val;



272                                numCarCurr = 1;



273                            } break;



274                            default:



275                                numCarCurr = 0;



276                                break;



277                            }



278                        } else {



279                            targetFramePtr = va_arg(argptr, char*);



280                            numCarCurr = (int)strlen(targetFramePtr);



281                        }



282



283                        if (numCarCurr < 0) {



284                            /* set error id if error or buffer overruns */



285                            errorId = TARGET_IO_BUFFER_TRUNC;



286                            bFinished = 1;



287                            expectedSize = 0;



288                            break;



289                        } else {



290                            /* we send the buffer like this */



291                            if ((expectedSize + numCarCurr) >= TARGET_IO_BUFFER_RES_SIZE) {



292                                /*keep this for after*/



293                                targetStdioTmpBuffIdx = numCarCurr;



294                                if (subFmt[subFmtIdx] != 's') {



295                                    targetFramePtr = &targetStdioTmpBuff[0];



296                                }



297                                bSendNow = 1;



298                            } else {



299                                if (subFmt[subFmtIdx] != 's') {



300                                    strncpy(&targetStdioBuffer[expectedSize],



301                                            &targetStdioTmpBuff[0], (size_t)numCarCurr);



302                                } else {



303                                    strncpy(&targetStdioBuffer[expectedSize], targetFramePtr,



304                                            (size_t)numCarCurr);



305                                    targetFramePtr = (char*)0;



306                                }



307                                expectedSize += (uint16_T)numCarCurr;



308                            }



309                        }



310                    } else {



311                        subFmtIdx++;



312                    }



313                    /* if we have not found but we have '\0', then we have a wrong input frame*/



314                } while ((!found) && (fmt[idx] != '\0'));



315



316                /* here we didn't found a correct format but this is the end of the fmt, we don't



317                   display this last part. this is not defined by the standard. This is an error



318                   case



319                  */



320                if ((fmt[idx] == '\0') && (!found)) {



321                    bFinished = 1;



322                    bSendNow = 1;



323                }



324                idx++;



325            } else if (fmt[idx] == '\0') {



326                bFinished = 1;



327                bSendNow = 1;



328            } else {



329                targetStdioBuffer[expectedSize] = fmt[idx];



330                expectedSize++;



331                if (expectedSize >= TARGET_IO_BUFFER_RES_SIZE) {



332                    bSendNow = 1;



333                }



334                idx++;



335            }



336        }



337



338        if (bSendNow && ((expectedSize > 0) || (errorId != TARGET_IO_SUCCESS))) {



339            if (expectedSize > 0) {



340                printSize = expectedSize + 1;



341                targetStdioBuffer[expectedSize] = '\0';



342                expectedSize = 0;



343            }



344



345            targetWritePrintf(targetFid, responseId, errorId, &targetStdioBuffer[0], printSize);



346



347            bSendNow = 0;



348        }



349



350        if ((targetStdioTmpBuffIdx > 0) && (!bFinished)) {



351            if (targetStdioTmpBuffIdx < TARGET_IO_BUFFER_RES_SIZE) {



352                strncpy(&targetStdioBuffer[0], targetFramePtr, (size_t)targetStdioTmpBuffIdx);



353                expectedSize = (uint16_T)targetStdioTmpBuffIdx;



354                targetStdioTmpBuffIdx = 0;



355                targetFramePtr = 0;



356



357            } else {



358                strncpy(&targetStdioBuffer[0], targetFramePtr, (size_t)TARGET_IO_BUFFER_RES_SIZE);



359                targetFramePtr += TARGET_IO_BUFFER_RES_SIZE;



360                targetStdioTmpBuffIdx -= TARGET_IO_BUFFER_RES_SIZE;



361                expectedSize = (uint16_T)TARGET_IO_BUFFER_RES_SIZE;



362



363                bSendNow = 1;



364            }



365        }



366



367    } while (!bFinished);



368}

391/* Forward a fopen back to the host.



392 *



393 * fid is used on target to act as a valid 



394 * "file descriptor".



395 */



396unsigned short targetFopen(const char * fileName) {



397   uint16_T fileNameSize;



398   MemUnit_T responseId = XIL_RESPONSE_FOPEN;



399



400   /* size of file name string */



401   fileNameSize = (uint16_T) strlen(fileName);



402



403   /* add one for the terminating null character */



404   fileNameSize++;



405 



406   /* check bounds and reset to 0 if necessary which will trigger an error 



407    * on the host */



408   if(fid >= TARGET_IO_MAX_FILE_ID) {



409      fid = 0;



410   }



411   /* send response id */



412   targetWriteDataWithErrorCheck(&responseId, sizeof(responseId));



413   /* send file id */



414   targetWriteDataWithErrorCheck((MemUnit_T *)&fid, sizeof(fid));



415   /* send size of file name string */



416   targetWriteDataWithErrorCheck((MemUnit_T *)&fileNameSize, sizeof(fileNameSize));



417#ifdef HOST_WORD_ADDRESSABLE_TESTING



418   /* in case the total size is bigger than the hostWordStdioBuffer, send in 



419    * multiple chunks */



420   {      



421      while (fileNameSize > 0) {



422         uint16_T transferSize = MIN(fileNameSize, TARGET_IO_BUFFER_SIZE);



423         /* send file name string */



424         targetWriteDataWithCharToMemUnitWidening(fileName, &hostWordStdioBuffer[0], (uint32_T)transferSize);   



425         fileNameSize -= transferSize;



426         fileName += transferSize;



427      }



428   }



429#else



430   /* send file name string */



431   targetWriteDataWithErrorCheck((const MemUnit_T *)fileName, fileNameSize);



432#endif



433



434   /* return fid, THEN increment it */



435   return (unsigned short) (fid++); /* unsigned short is at least 16-bits



436                                     * wide. See header file for more



437                                     * information on why this cast is



438                                     * required. */



439}
